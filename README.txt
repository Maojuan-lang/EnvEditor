中文：

# type:4
# 说明：这行会被丢弃
# 页面名称
any_var_name=0

# type:1
# 说明：变量为0或1
# 可以添加任意行描述
# 单选框名称
var=0

# type:2
# value:1.apple.2.banana.3.peach
# 可以添加任意行描述
# 下拉框名称
var2=1

# type:3
# value:1.apple.2.banana.3.peach
# 说明：至少有一个被选中，如果都没选中，默认保存第一个
# 可以添加任意行描述
# 多选框名称
var3=1

# type:5
# value:1.apple.2.banana.3.peach
# 说明：经典队列
# 队列名称
var5=1.2.3

English：

# type:4
# description: this will be discard
# tab name
any_var_name=0

# type:1
# description: radio_button，var's value is 0 or 1
# description any lines
# radio_button_name
var=0

# type:2
# value:1.apple.2.banana.3.peach
# description any lines
# ComboBox_name
var2=1

# type:3
# value:1.apple.2.banana.3.peach
# description : at least one was selected, default was the first one
# description any lines
# CheckBox_name
var3=1

# type:5
# value:1.apple.2.banana.3.peach
# description: the classical queue
# queue_name
var5=1.2.3
